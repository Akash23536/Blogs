How to build a counter app using JavaScript and React.js

Counter App: The Counter App will have an increment, decrement, and reset button. And a display to show the number.


Counter app using Dom Manipulation:-

                
             // Html code:

                  <h1> Counter </h1>
                  <h1 id = "counter"> 0 </h1>
                  <button id = "inc"> increment </button>
                  <button id = "dec"> decrement </button>
                  <button id = "reset"> reset </button>

             // JavaScript code:

                  let value = 0;
                  const count = document.getElementById("counter");
                  const inc = document.getElementById("inc");
                  const dec = document.getElementById("dec");
                  const reset = document.getElementById("reset");
                    inc.onclick = function () {
                    value++;
                    count.innerHTML = `${value}`;
                  };
                    dec.onclick = () => {
                    value--;
                    count.innerHTML = `${value}`;
                  };
                    reset.onclick = function () {
                    value = 0;
                    count.innerHTML = `${value}`;
                  };

Counter app using react useState hook

                
                  import { useState } from 'react';

                  function App() {
                    const [count, setCount] = useState(0);
                    let increase = () => {
                      setCount(count + 1);
                    };
                    let decrease = () => {
                      setCount(count - 1);
                    };
                    let reset = () => {
                      setCount(0);
                    };

                    return (
                      <>
                        <h1>Counter</h1>
                        <h1>{count}</h1>
                        <button onClick={increase}>increase</button>
                        <button onClick={decrease}>decrease</button>
                        <button onClick={reset}>reset</button>
                      </>
                    );
                  }
                  export default App;
