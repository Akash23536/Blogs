How to build a Jokes app using JavaScript and React.js

Jokes App: The App will display random joke and a Button to refresh the jokes


Jokes App using Dom Manipulation and fetch api:-

     
// Html code:

<div id="jokeElement">Loading...</div>
<button id="refreshButton">Refresh</button>

// JavaScript code:

const jokeElement = document.getElementById("jokeElement");
const refreshButton = document.getElementById("refreshButton");

async function fetchJoke() {
try {
const response = await fetch("https://icanhazdadjoke.com/slack");
const jokeData = await response.json();
const jokeText = jokeData.attachments[0].text;
jokeElement.innerHTML = jokeText;
} catch (error) {
console.error("Error fetching joke:", error);
jokeElement.innerHTML = "Failed to fetch a joke. Please try again later.";
}
}

refreshButton.addEventListener("click", fetchJoke);
fetchJoke();

jokes App using react useState, useEffect hooks and fetch api:-

  
  import React, { useState, useEffect } from 'react';
  
  function App() {
  const [joke, setJoke] = useState('Loading...');

  const fetchJoke = async () => {
  try {
    const response = await fetch("https://icanhazdadjoke.com/slack");
    const jokeData = await response.json();
    const jokeText = jokeData.attachments[0].text;
    setJoke(jokeText);
  } catch (error) {
    console.error("Error fetching joke:", error);
    setJoke("Failed to fetch a joke. Please try again later.");
  }
  };

  useEffect(() => {
    fetchJoke();
  }, []);

  return (
  <div>
  <div id="jokeElement">{joke}</div>
  <button id="refreshButton" onClick={fetchJoke}>Refresh</button>
  </div>
  );
  }
  export default App;
