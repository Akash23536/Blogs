How to build a Background color Switcher app using JavaScript and React.js

Background color Switcher App: The App will have a colored button. When a click event is done on it, the background color changes to the same as the button color.


Background color Switcher App using Dom Manipulation:-

        
          // HTML code:
          <h1> Background Color Switcher </h1>
          <span id="grey"> grey </span>
          <span id="green"> green </span>
          <span id="blue"> blue </span>
          <span id="yellow"> yellow </span>

          // JavaScript code:
          const grey = (document.getElementById("grey").onclick = function () {
            document.body.style.backgroundColor = "grey";
          });
          const green = (document.getElementById("green").onclick = () => {
            document.body.style.backgroundColor = "green";
          });
          const blue = (document.getElementById("blue").onclick = function () {
            document.body.style.backgroundColor = "blue";
          });
          const yellow = (document.getElementById("yellow").onclick = () => {
            document.body.style.backgroundColor = "yellow";
          });

Background color Switcher App using React useState hook:-

        
          import { useState } from 'react';
          function App() {
            const [color, setColor] = useState("black");
            function bgYellow() { setColor("yellow"); }
            return (
              <div style={{ backgroundColor: color }}>
                <h1> Background Color Switcher </h1>
                <span onClick={()=>setColor("red")}> red </span>
                <span onClick={()=>setColor("green")}> green </span>
                <span onClick={()=>setColor("blue")}> blue </span>
                <span onClick={bgYellow}> yellow </span>
              </div>
            );
          }
          export default App;
